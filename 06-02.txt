クエリ

SELECT
	dpt
FROM
	Students
GROUP BY
	dpt
HAVING
	SUM(
		CASE
			-- 2018年9月30日以前なら1
			WHEN sbmt_date < '2018-09-30' THEN 1
		-- そのほかの場合、0
		ELSE 0
		END
	)
	= COUNT(*)
;
