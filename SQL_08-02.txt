クエリ
SELECT
	employee,
	(COUNT(NULLIF(child_1, '')) +
	COUNT(NULLIF(child_2, '')) +
	COUNT(NULLIF(child_3, ''))) AS child_cnt
FROM
	Personnel
GROUP BY
	employee
ORDER BY
	child_cnt DESC
;
