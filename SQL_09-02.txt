クエリ
SELECT
	ES.emp
FROM
	EmpSkills ES
LEFT OUTER JOIN
	Skills S
ON
	ES.skill = S.skill
GROUP BY
	ES.emp
HAVING
	COUNT(ES.skill) = (SELECT COUNT(skill) FROM skills)
AND
	COUNT(S.skill) =  (SELECT COUNT(skill) FROM skills)
;
