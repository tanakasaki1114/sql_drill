クエリ
SELECT
	ES.emp
FROM
	EmpSkills ES
LEFT OUTER JOIN
	Skills S
ON
	ES.skill = S.skill
GROUP BY
	ES.emp
HAVING
	COUNT(ES.skill) = (SELECT COUNT(skill) FROM skills)
AND
	COUNT(CASE 
		WHEN ES.skill IN (SELECT 
					skill 
						FROM 
							-- 1を返す
							Skills) THEN 1 END) = (SELECT 
											COUNT(skill) 
													FROM
														Skills)
;
