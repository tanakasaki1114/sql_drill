クエリ

SELECT
	b.title,
	COUNT(r.isbn)
FROM
	books AS b
LEFT JOIN
	rental AS r
ON
	b.isbn = r.isbn
GROUP BY
	b.title
ORDER BY
	r.isbn DESC
;
